from openai import OpenAI client = OpenAI( api_key="YOUR_GUIJIAPI_KEY", base_url="https://api.guijiapi.net/v1", ) # 任意模型，改 model 名即可切换 resp = client.chat.completions.create( model="claude-sonnet-4-6", # 或 gpt-4o / gemini-2.5-pro … messages=[{"role": "user", "content": "你好！"}], ) print(resp.choices[0].message.content)
Python · Anthropic SDK
import anthropic client = anthropic.Anthropic( api_key="YOUR_GUIJIAPI_KEY", base_url="https://api.guijiapi.net", ) msg = client.messages.create( model="claude-sonnet-4-6", max_tokens=1024, messages=[{"role": "user", "content": "你好！"}], ) print(msg.content[0].text)
